 42def is_included(
 43    path: str,
 44    only_process_files: set[str] | None = None,
 45) -> bool:
 46    """Checks if a file is excluded by not being mentioned in
 47    the only process files list. Matches only the basename of
 48    the files.
 49
 50    :param path: The path to be checked.
 51
 52    :param only_process_files: The include file list to check the path for.
 53        If `None`, the file will be included. Defaults to `None`.
 54
 55    :return: `True` if the path should be included, `False` otherwise.
 56    """
 57    if only_process_files:
 58        # Only compares the basename.
 59        basename = os.path.basename(path)
 60        for file in only_process_files:
 61            basename2 = os.path.basename(file)
 62            if basename == basename2:
 63                return True
 64        return False
 65    else:
 66        # Always include the file if no set was given.
 67        return True

 70def update_cache(
 71    root_dir: str = ".",
 72    only_process_files: set[str] | None = None,
 73    quiet: bool = False,
 74) -> None:
 75    """Updates the cache.
 76
 77    :param root_dir: The root path. In this directory, the cache directory will be located,
 78        and it should also be the root of all images to be analysed.
 79
 80    :param only_process_files: A set of image paths to be processed. If `None`, all found images
 81    will be classified.
 82
 83    :param quiet: Whether additional verbose output should be generated.
 84    """
 85    # Checks for correct parameters
 86    root_dir = os.path.abspath(root_dir)
 87    if not os.path.exists(root_dir):
 88        raise FacesException(f"Root directory ({root_dir}) does not exist.")
 89    if not os.path.isdir(root_dir):
 90        raise FacesException(f"Path ({root_dir}) is not a directory.")
 91    pass
 92
 93    if not quiet:
 94        print("[green]++ Update cache ++[/green]")
 95
 96    image_files_root = find_image_files(root_dir, for_albums=False)
 97    for image in image_files_root:
 98        # Only process files which are not excluded
 99        if not is_included(image, only_process_files):
100            continue
101
102        # Generate an MD5 hash, as this is used for the cache file name
103        with open(image, "rb") as f:
104            image_hash = hashlib.md5(f.read()).hexdigest()
105
106        encodings_dir = os.path.join(
107            root_dir, FACES_CACHE_DIR_NAME, image_hash
108        )
109        # Only re-calculates the encodings for images which were not classified
110        # in an earlier run.
111        if not os.path.exists(encodings_dir):
112            pathlib.Path(encodings_dir).mkdir(parents=True, exist_ok=True)
113            if not quiet:
114                print(
115                    f"  [blue]++ Calculating image '{os.path.basename(image)}'"
116                    " face encodings ++[/blue]"
117                )
118
119            # Calculate the face encodings
120            encodings_data = get_face_encodings(image, quiet=quiet)
121
122            # Serialise the face encodings
123            for idx, encoding in enumerate(encodings_data):
124                if not quiet:
125                    print(
126                        f"    [blue]++ Writing image '{os.path.basename(image)}'"
127                        f" face encoding {idx + 1}/{len(encodings_data)} ++[/blue]"
128                    )
129                serialized_as_json = json.dumps(
130                    pickle.dumps(encoding).decode("latin-1")
131                )
132                # Store the individual encoding in a filename that is the MD5 hash of the
133                # encoding itself.
134                md5 = str(
135                    hashlib.md5(serialized_as_json.encode("utf-8")).hexdigest()
136                )
137                output_file = os.path.join(encodings_dir, md5 + ".encoding")
138                with open(output_file, "w") as f:
139                    f.write(serialized_as_json)

142def clear_cache(root_dir: str = ".", quiet: bool = False) -> None:
143    """Clears the cache.
144
145    :param root_dir: The root path. In this directory, the cache directory will be searched.
146
147    :param quiet: Whether additional verbose output should be generated.
148    """
149    # Check the parameters
150    root_dir = os.path.abspath(root_dir)
151    if not os.path.exists(root_dir):
152        raise FacesException(f"Root directory ({root_dir}) does not exist.")
153    if not os.path.isdir(root_dir):
154        raise FacesException(f"Path ({root_dir}) is not a directory.")
155    try:
156        # Remove the cache directory recursively if it exists
157        shutil.rmtree(os.path.join(root_dir, CACHE_BASE_NAME))
158        if not quiet:
159            print(f"[green]++ Cleared cache directory ({root_dir}) ++[/green]")
160    except FileNotFoundError:
161        if not quiet:
162            print(f"[red]++ No previous cache found ({root_dir}) ++[/red]")

165def handle_delete_old(
166    albums_root_dir: str,
167    only_process_files: set[str] | None = None,
168    quiet: bool = False,
169    dry_run: bool = False,
170) -> None:
171    """Internal function to handle the deletion of old files in albums.
172
173    :param albums_root_dir: The root path containing the albums.
174        Albums will be searched in the whole hierarchy.
175
176    :param only_process_files: A set of image paths to be processed. If `None`, all found images
177    will be considered for deletion.
178
179    :param quiet: Whether additional verbose output should be generated.
180
181    :param dry_run: Whether to only print the actions which would be executed.
182    """
183    if not quiet:
184        print("[green]++ Check for old files to remove ++[/green]")
185
186    # Find all directories which are configured to be used with `CutyX`
187    album_dirs = find_album_dirs(albums_root_dir)
188
189    if only_process_files:
190        # Do a matching on the base name to check, whether the file should be removed
191        image_files_albums = find_image_files(albums_root_dir, for_albums=True)
192        for image_album_file in image_files_albums:
193            if is_included(image_album_file, only_process_files):
194                if not os.path.isdir(image_album_file):
195                    if not quiet:
196                        print(
197                            "  [blue]++ Remove previously classified image "
198                            f"'{os.path.basename(image_album_file)}' ++[/blue]"
199                        )
200                    if not dry_run:
201                        os.remove(image_album_file)
202    else:
203        # Remove all non-hidden files in a found album directory
204        for album_dir in album_dirs:
205            album_dir_files = [
206                file
207                for file in os.listdir(album_dir)
208                if not file.startswith(".")
209            ]
210            for album_dir_file in album_dir_files:
211                file_to_remove = os.path.join(album_dir, album_dir_file)
212                if not os.path.isdir(file_to_remove):
213                    if not quiet:
214                        print(
215                            "  [blue]++ Remove previously classified image "
216                            f"'{album_dir_file}' ({os.path.basename(album_dir)}) ++[/blue]"
217                        )
218                    if not dry_run:
219                        os.remove(file_to_remove)

222def process_directory(
223    root_dir: str = ".",
224    albums_root_dir: str = ".",
225    dry_run: bool = False,
226    delete_old: bool = True,
227    symlink: bool = True,
228    use_cache: bool = True,
229    quiet: bool = False,
230    only_process_files: set[str] | None = None,
231) -> None:
232    """The main logic of **CutyX**.
233
234    This function will process the files, classify them, and write the results to the
235    album directories.
236
237    :param root_dir: The root path containing the images to be organised.
238        Images will be searched in the whole hierarchy.
239
240    :param albums_root_dir: The root path containing the albums.
241        Albums will be searched in the whole hierarchy.
242
243    :param dry_run: Whether to only print the actions which would be executed.
244
245    :param delete_old: Whether to delete old images in the album directories,
246        before processing the new ones.
247
248    :param symlink: Whether to symlink from the album folder instead of copying it.
249
250    :param use_cache: Whether the cache should be used.
251
252    :param quiet: Whether additional verbose output should be generated.
253
254    :param only_process_files: A set of image paths to be processed. If `None`, all found images
255    will be considered for processing.
256    """
257    handle_dry_run(dry_run)
258
259    # Check only-process files for validity
260    if only_process_files:
261        only_process_files = {os.path.abspath(f) for f in only_process_files}
262        for f in only_process_files:
263            check_valid_image(f)
264
265    # Update the cache if the cache should be used
266    if not dry_run and use_cache:
267        update_cache(root_dir, only_process_files=only_process_files)
268
269    root_dir = os.path.abspath(root_dir)
270
271    # Handle deletion of old files
272    if delete_old:
273        handle_delete_old(
274            albums_root_dir,
275            only_process_files=only_process_files,
276            quiet=quiet,
277            dry_run=dry_run,
278        )
279
280    # Run the actual processing
281    handle_process_files(
282        root_dir,
283        albums_root_dir,
284        only_process_files=only_process_files,
285        quiet=quiet,
286        use_cache=use_cache,
287        symlink=symlink,
288        dry_run=dry_run,
289    )

292def handle_process_files(
293    root_dir: str,
294    albums_root_dir: str,
295    only_process_files: set[str] | None = None,
296    quiet: bool = False,
297    use_cache: bool = False,
298    symlink: bool = False,
299    dry_run: bool = False,
300) -> None:
301    """The main processing and classification logic.
302
303    :param root_dir: The root path containing the images to be organised.
304        Images will be searched in the whole hierarchy.
305
306    :param albums_root_dir: The root path containing the albums.
307        Albums will be searched in the whole hierarchy.
308
309    :param dry_run: Whether to only print the actions which would be executed.
310
311    :param symlink: Whether to symlink from the album folder instead of copying it.
312
313    :param use_cache: Whether the cache should be used.
314
315    :param quiet: Whether additional verbose output should be generated.
316
317    :param only_process_files: A set of image paths to be processed. If `None`, all found images
318    will be considered for processing.
319    """
320    num_processed = 0
321
322    # Search for all images
323    album_dirs = find_album_dirs(albums_root_dir)
324    image_files_root = find_image_files(root_dir, for_albums=False)
325
326    if not quiet:
327        if not image_files_root:
328            print("[red]++ Found no images ++[/red]")
329        else:
330            print(f"[green]++ Found {len(image_files_root)} images ++[/green]")
331
332    # Process the albums
333    for album_dir in album_dirs:
334        if not quiet:
335            print(
336                f"[green]++ Process album directory '{os.path.basename(album_dir)} ++[/green]"
337            )
338        for file in image_files_root:
339            do_process = True
340            if only_process_files:
341                if file not in only_process_files:
342                    do_process = False
343            cache_root_dir: str | None = root_dir
344            if not use_cache:
345                cache_root_dir = None
346            if do_process and any_matches(
347                [
348                    ("name", lambda: name_matches(file, album_dir)),
349                    (
350                        "face",
351                        lambda: face_matches(
352                            file,
353                            album_dir,
354                            cache_root_dir=cache_root_dir,
355                            quiet=quiet,
356                        ),
357                    ),
358                ],
359                quiet=quiet,
360            ):
361                # We got a match => Copy or symlink the file to the albums folder
362                num_processed += 1
363                if symlink:
364                    if not quiet:
365                        print(
366                            f"  [blue]++ Symlink '{os.path.basename(file)}' -> "
367                            f"'{os.path.basename(album_dir)}/' ++[/blue]"
368                        )
369                    if not dry_run:
370                        mksymlink(
371                            file,
372                            os.path.join(album_dir, os.path.basename(file)),
373                        )
374                else:
375                    if not quiet:
376                        print(
377                            f"  [blue]++ Copy '{os.path.basename(file)}' -> "
378                            f"'{os.path.basename(album_dir)}/' ++[/blue]"
379                        )
380                    if not dry_run:
381                        copyfile(
382                            file,
383                            os.path.join(album_dir, os.path.basename(file)),
384                        )
385
386    # Raises an exception if albums are configured, but not a single match
387    # was found.
388    if num_processed == 0 and image_files_root and album_dirs:
389        raise FacesException("No images were matching for any album.")

392def find_album_dirs(root_dir: str) -> list[str]:
393    """Searches hierarchically for all configured album directories.
394
395    :param root_dir: The root path containing the albums.
396        Albums will be searched in the whole hierarchy.
397
398    :return: A `list` of all found album directories.
399    """
400    dirs: list[str] = []
401    for root, dnames, _ in os.walk(
402        root_dir, topdown=True, onerror=None, followlinks=True
403    ):
404        # Search for albums in the current and child directories
405        for d in [root] + dnames:
406            dpath = os.path.join(root_dir, d)
407            faces_path = os.path.join(dpath, FACES_DIR_NAME)
408            # If a faces dir is in the directory, it is considered as an album
409            if os.path.exists(faces_path):
410                dirs.append(dpath)
411    return dirs

414def find_image_files(root_dir: str, for_albums: bool = False) -> list[str]:
415    """Searches hierarchically for all images.
416
417    :param root_dir: The root path containing the images.
418        Images will be searched in the whole hierarchy.
419
420    :param for_albums: Whether the images should return images from albums directories
421        or only from non-album directories.
422
423    :return: A `list` of all found images.
424    """
425    images: list[str] = []
426    for root, _, fnames in os.walk(
427        root_dir, topdown=True, onerror=None, followlinks=True
428    ):
429        for fname in fnames:
430            dpath = os.path.join(root_dir, root)
431            faces_path = os.path.join(dpath, FACES_DIR_NAME)
432            include_file = False
433            if os.path.exists(faces_path):
434                if for_albums:
435                    include_file = True
436            else:
437                if not for_albums:
438                    include_file = True
439            if include_file:
440                fpath = os.path.join(dpath, fname)
441                if is_valid_image(fpath):
442                    images.append(fpath)
443    return images

495def match_names(
496    album_dir: str,
497    text: str,
498    dry_run: bool = False,
499    use_regex: bool = False,
500    use_fuzzy: bool = False,
501    fuzzy_min_ratio: int = 60,
502    rule_name: str | None = None,
503    quiet: bool = False,
504) -> None:
505    """Adds a rule to match files by name.
506
507    :param album_dir: The album to add the training data to.
508
509    :param text: The text to match by.
510
511    :param use_regex: Whether regular expressions should be used for matching.
512
513    :param use_fuzzy: Does a fuzzy match.
514
515    :param fuzzy_min_ratio: Minimum ratio to score a fuzzy match.
516
517    :param dry_run: Whether to only print the actions which would be executed.
518
519    :param rule_name: A name for the rule in the hidden album directory.
520        Can be used to make it easier to remove rules from the classification later.
521
522    :param quiet: Whether additional verbose output should be generated.
523    """
524    handle_dry_run(dry_run)
525
526    if use_regex:
527        if use_fuzzy:
528            raise FacesException("Fuzzy and Regex cannot be combined.")
529        try:
530            re.compile(text)
531        except re.error as e:
532            raise FacesException(f"Invalid regular expression: '{str(e)}'")
533
534    names_data = {
535        "text": text,
536        "use_regex": use_regex,
537        "use_fuzzy": use_fuzzy,
538        "fuzzy_min_ratio": fuzzy_min_ratio,
539    }
540
541    if not quiet:
542        print(
543            f"[green]++ Creating name rule '{text}' "
544            f"(regex={use_regex}, fuzzy={use_fuzzy}, fuzzy_ratio={fuzzy_min_ratio}) ++[/green]"
545        )
546
547    if rule_name:
548        output_file_name = rule_name + NAMES_FILE_EXT
549    else:
550        text_hash = hashlib.md5(text.encode("utf-8")).hexdigest()
551        output_file_name = text_hash + NAMES_FILE_EXT
552
553    output_file = os.path.join(album_dir, FACES_DIR_NAME, output_file_name)
554
555    if not dry_run:
556        pathlib.Path(os.path.dirname(output_file)).mkdir(
557            parents=True, exist_ok=True
558        )
559        with open(output_file, "w") as f:
560            f.write(json.dumps(names_data))

563def match_faces(
564    album_dir: str,
565    training_image_path: str,
566    dry_run: bool = False,
567    training_data_prefix: str | None = None,
568    quiet: bool = False,
569) -> None:
570    """Adds training data for a face classification to an album.
571
572    :param album_dir: The album to add the training data to.
573
574    :param training_image_path: The training data image. All faces found in this image will
575        be used to classify images as valid for this album.
576
577    :param dry_run: Whether to only print the actions which would be executed.
578
579    :param training_data_prefix: A prefix for training data of this image in the hidden
580        album directory. Can be used to make it easier to remove faces from the classification
581        later.
582
583    :param quiet: Whether additional verbose output should be generated.
584    """
585    handle_dry_run(dry_run)
586    check_valid_image(training_image_path)
587
588    if not quiet:
589        print(
590            f"[green]++ Calculate face encodings '{training_image_path}' ++[/green]"
591        )
592    encodings = get_face_encodings(training_image_path, quiet=quiet)
593
594    if len(encodings) == 0:
595        raise FacesException(
596            f"No face recognised in image '{training_image_path}."
597        )
598
599    with open(training_image_path, "rb") as f:
600        image_hash = hashlib.md5(f.read()).hexdigest()
601
602    output_training_dir_name = image_hash + TRAINING_IMAGE_DIR_EXT
603    if training_data_prefix:
604        output_training_dir_name = (
605            training_data_prefix + "-" + output_training_dir_name
606        )
607
608    output_dir = os.path.join(
609        album_dir, FACES_DIR_NAME, output_training_dir_name
610    )
611    if not quiet:
612        print(
613            f"  [blue]++ Create training data directory '{output_dir}' ++[/blue]"
614        )
615    # Re-generate the training data
616    if not dry_run:
617        try:
618            shutil.rmtree(output_dir)
619        except FileNotFoundError:
620            pass
621        pathlib.Path(output_dir).mkdir(parents=True, exist_ok=True)
622
623    # Write all found face encodings
624    for idx, encoding in enumerate(encodings):
625        if not quiet:
626            print(
627                f"  [blue]++ Writing face encoding {idx + 1}/{len(encodings)} ++[/blue]"
628            )
629        serialized_as_json = json.dumps(
630            pickle.dumps(encoding).decode("latin-1")
631        )
632        md5 = str(hashlib.md5(serialized_as_json.encode("utf-8")).hexdigest())
633        output_file = os.path.join(output_dir, md5 + ".encoding")
634        if not dry_run:
635            with open(output_file, "w") as f:
636                f.write(serialized_as_json)
637
638    # Generates a symlink to the training image to make it easier to remove it later.
639    symlink_path = image_hash + TRAINING_IMAGE_SRC_EXT
640    if training_data_prefix:
641        symlink_path = training_data_prefix + "-" + symlink_path
642    symlink_path = os.path.join(album_dir, FACES_DIR_NAME, symlink_path)
643    if not quiet:
644        print("  [blue]++ Create symlink to training image ++[/blue]")
645    if not dry_run:
646        try:
647            os.remove(symlink_path)
648        except FileNotFoundError:
649            pass
650        mksymlink(os.path.abspath(training_image_path), symlink_path)

653def is_valid_image(image_path: str) -> bool:

662def check_valid_image(image_path: str) -> None:
663    """Requires validity for an image. Throws an exception otherwise."""
664    if not os.path.exists(image_path):
665        raise FacesException(f"Path '{image_path}' does not exist.")
666    if not os.path.isfile(image_path):
667        raise FacesException(f"Path '{image_path}' is no file.")
668    if not (
669        image_path.lower().endswith(".jpg")
